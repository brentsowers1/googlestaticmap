def to_s
  if latitude && longitude
    "#{CGI.escape(latitude.to_s)},#{CGI.escape(longitude.to_s)}"
  elsif address
    CGI.escape(address)
  else
    raise Exception.new("Need to set either latitude and longitude, or address")
  end
end
